		$(document).on("click",".search_btn",function()
		{
			var title = $('#title_name').val();
			var procuring_entity = $('.procuring_entity').val();
			var publish_from_date = $('.publish_from_date').val();
			var publish_to_date = $('.publish_to_date').val();
			var closing_from_date = $('.closing_from_date').val();
			var closing_to_date = $('.closing_to_date').val();
			$.ajax({
				url : "https://www.bwdb.gov.bd/haor/awarded_tender_search",
				data: {
						'title': title,
						'publish_from_date': publish_from_date,
						'publish_to_date':publish_to_date,
						'closing_from_date': closing_from_date,
						'closing_to_date':closing_to_date,
						'procuring_entity': procuring_entity,
					},
				dataType: "html",
				type: "POST",
				beforeSend: function() {
					$('.loading_div').show();
				},
				success: function(data){
					$('.loading_div').hide();
					$('.tender_list').html(data);
				},
				error: function(data){
					$('.loading_div').hide();
				}
			});
		});
